class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        for(int i = 0; i< nums.length; i++) {
            if(map.containsKey(target - nums[i])) {
                return new int[] {map.get(target-nums[i]),i};
            }
            map.put(nums[i], i);
        }
        throw new IllegalArgumentException("No two sum solution");
    }
}

class Solution {
    public int[] twoSum(int[] nums, int target) {
        int i, j;
        for (i = 0; i < nums.length; i++) {
            for (j = i+1; j < nums.length; j++) {
                if (nums[i] + nums[j] == target) {
                    return new int[]{i, j};
                }
            }
        }
        return null;
    }
}

class Solution {
    public boolean isPalindrome(int x) {
        // 特殊情况：
        // 如上所述，当 x < 0 时，x 不是回文数。
        // 同样地，如果数字的最后一位是 0，为了使该数字为回文，
        // 则其第一位数字也应该是 0
        // 只有 0 满足这一属性
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }

        int revertedNumber = 0;
        while (x > revertedNumber) {
            revertedNumber = revertedNumber * 10 + x % 10;
            x /= 10;
        }

        // 当数字长度为奇数时，我们可以通过 revertedNumber/10 去除处于中位的数字。
        // 例如，当输入为 12321 时，在 while 循环的末尾我们可以得到 x = 12，revertedNumber = 123，
        // 由于处于中位的数字不影响回文（它总是与自己相等），所以我们可以简单地将其去除。
        return x == revertedNumber || x == revertedNumber / 10;
    }
}

class Solution {
    public boolean isPalindrome(int x) {
        if (x < 0) return false;
        String strX = Integer.toString(x);
        int length = strX.length();
        for (int i = 0; i < length / 2; i++) {
            char left = strX.charAt(i);
            char right = strX.charAt(length - 1 - i);
            
            if (left != right) return false;
        }
        return true;
    }
}

class Solution {
    Map<Character, Integer> symbolValues = new HashMap<Character, Integer>() {{
        put('I', 1);
        put('V', 5);
        put('X', 10);
        put('L', 50);
        put('C', 100);
        put('D', 500);
        put('M', 1000);
    }};

    public int romanToInt(String s) {
        int ans = 0;
        int n = s.length();
        for (int i = 0; i < n; ++i) {
            int value = symbolValues.get(s.charAt(i));
            if (i < n - 1 && value < symbolValues.get(s.charAt(i + 1))) {
                ans -= value;
            } else {
                ans += value;
            }
        }
        return ans;
    }
}

class Solution {
    public String longestCommonPrefix(String[] strs) {
        // 1. 特殊情况：如果数组为空，直接返回空字符串
        if (strs == null || strs.length == 0) {
            return "";
        }

        // 2. 核心：把第一个字符串当作【初始的最长公共前缀】
        // 比如：["flower","flow","flight"]
        // prex 一开始 = "flower"
        String prex = strs[0];

        // 3. 遍历数组中【剩下的每一个字符串】
        // 从第 2 个开始，一个个去匹配当前前缀 prex
        for (int i = 1; i < strs.length; i++) {

            // 4. 关键：indexOf(prex) != 0
            // 意思：当前字符串【不是以 prex 开头】，就进入循环
            // indexOf(xxx) == 0 才代表【以xxx开头】
            while (strs[i].indexOf(prex) != 0) {

                // 5. 不匹配？就把前缀【删掉最后一个字符】
                // 比如 "flower" → "flowe" → "flow" → "flo" → "fl"
                prex = prex.substring(0, prex.length() - 1);

                // 6. 如果前缀缩到空了，直接返回空，没有公共前缀
                if (prex.isEmpty()) {
                    return "";
                }
            }
        }

        // 7. 遍历完所有字符串，剩下的 prex 就是【最长公共前缀】
        return prex;
    }
}

class Solution {
    public String longestCommonPrefix(String[] strs) {
        // 特殊情况：数组为空直接返回空
        if (strs == null || strs.length == 0) {
            return "";
        }

        String pub = "";
        int j = 0;

        while(j < strs[0].length()) {
            // 标记数组第一个字符串的第一个字符
            char target = strs[0].charAt(j);
            // 遍历数组
            for (int i = 1; i < strs.length; i++) {
                if (j >= strs[i].length()) return pub;
                if (target != strs[i].charAt(j)) {
                    return pub;
                }
            }
            pub += target;
            j++;
        }
        return pub;
    }
}

class Solution {
    public boolean isValid(String s) {
        int len = s.length();
        if (len % 2 == 1) {
            // 若长度是奇数则一定不匹配
            return false;
        }

        Map<Character, Character> pairs = new HashMap<Character, Character>() {{
            put(')', '(');
            put(']', '[');
            put('}', '{');
        }};

        // 创建一个栈
        Deque<Character> stack = new LinkedList<Character>();
        for (int i = 0; i < len; i++) {
            char ch = s.charAt(i);
            // 判断获取到的括号是不是右括号
            if (pairs.containsKey(ch)) {
                // 若是右括号则开始匹配
                if (stack.isEmpty() || stack.peek() != pairs.get(ch)) {
                    // 若栈为空那当然就没得匹配；
                    // 若不为空则判断当前栈顶（离它最近的左括号）是否是对应的左括号
                    // 若不匹配则返回false
                    return false;
                } else {
                    // 若成功匹配就出栈
                    stack.pop();
                }
            } else {
                // 若是左括号则入栈
                stack.push(ch);
            }
        }
        // 若栈为空，说明全部匹配后出栈了
        return stack.isEmpty();
    }
}

class Solution {
    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        } else if (list2 == null) {
            return list1;
        } else if (list1.val < list2.val) {
            list1.next = mergeTwoLists(list1.next, list2);
            return list1;
        } else {
            list2.next = mergeTwoLists(list1, list2.next);
            return list2;
        }
    }
}
